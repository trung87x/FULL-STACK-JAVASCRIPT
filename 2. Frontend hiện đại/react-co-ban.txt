function ProfileCard({ name, age }) {
  return (
    <div className="p-4 bg-white rounded shadow-md mb-4">
      <h2 className="text-xl font-bold text-blue-600">👤 Hồ sơ cá nhân</h2>
      <p>Tên: <strong>{name}</strong></p>
      <p>Tuổi: <strong>{age}</strong></p>
    </div>
  );
}

export default ProfileCard;


=======================================
2. src/components/Product.jsx
=======================================

function Product({ title, price }) {
  const handleBuy = () => {
    alert(`🛒 Bạn vừa mua: ${title} - Giá ${price} VNĐ`);
  };

  return (
    <div className="p-4 bg-white rounded shadow-md">
      <h3 className="text-lg font-semibold text-gray-800">{title}</h3>
      <p className="text-gray-600 mb-2">Giá: {price} VNĐ</p>
      <button
        onClick={handleBuy}
        className="bg-green-600 text-white px-3 py-1 rounded hover:bg-green-700 transition"
      >
        Mua
      </button>
    </div>
  );
}

export default Product;

import './index.css'
